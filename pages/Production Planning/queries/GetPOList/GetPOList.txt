SELECT
       i.name                        AS item,
       SUM(op.quantity)::int         AS quantity,
       /* pick a URL per item name */
       CASE i.name
            WHEN 'Novak racket'     THEN 'https://tenniscompanion.org/wp-content/uploads/2023/09/novak-djokovis-actual-racquet-head-pt-346.1.png'
            WHEN 'chair'            THEN 'https://www.kingliving.my/pub/media/catalog/product/a/u/aura_occ_1_1.jpg'
            WHEN 'chair leg'        THEN 'https://encrypted-tbn0.gstatic.com/images?q=tbn:ANd9GcTe7Df1Bu8GpFbIiYsaYnVMBs51yNoGwFngkZhYkDSYmp1eRt9tJ-2b4wMg6uG0pZjYY-0&usqp=CAU'
            WHEN 'cushion'          THEN 'https://thumbs.dreamstime.com/b/realistic-soft-pillow-sweet-dream-poster-white-pillows-flying-feathers-night-sky-clouds-pithy-vector-bed-relaxing-268445267.jpg'
            WHEN 'screws'           THEN 'https://i.pinimg.com/736x/f5/af/ba/f5afba57c119308a232f87b111c3c71e.jpg'
            WHEN 'varnished chair'  THEN 'https://cdn1.npcdn.net/image/165208264256ef9c94464f43567b56debc43c8c188.jpg?md5id=9866b8a83d35abdd89ed76d565d71f75&new_width=1150&new_height=2500&w=-62170009200'
            WHEN 'wooden beam'      THEN 'https://media.istockphoto.com/id/934317806/photo/structure-of-a-wooden-house-under-construction.jpg?s=612x612&w=0&k=20&c=BhSeH3TjzRgzT-YQ9rNj3P6Tv0kBnFoRYdGGSyZcwkM='
            ELSE 'https://encrypted-tbn0.gstatic.com/images?q=tbn:ANd9GcRkK98VBjmf1Q6_3SC9Nmz8CILkBdm1BUiFLg&s'
       END                           AS img
FROM   public.operationplan op
JOIN   public.item          i ON i.name = op.item_id
GROUP  BY i.name
ORDER  BY i.name;
