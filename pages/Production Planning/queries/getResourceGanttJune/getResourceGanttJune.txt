SELECT
  opr.resource_id AS resource,
  op.item_id,
  op.reference,
  op.startdate,
  op.enddate,
  op.status,
  op.quantity
FROM
  public.operationplan op
JOIN
  public.operationplanresource opr ON op.reference = opr.operationplan_id
WHERE
  op.startdate >= '2019-06-01' AND op.enddate < '2019-07-01'
ORDER BY
  opr.resource_id, op.startdate;
