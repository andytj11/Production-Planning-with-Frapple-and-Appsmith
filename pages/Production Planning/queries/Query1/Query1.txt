/* tasks that start in June-2019 */
SELECT
    r.name                      AS resource,          -- e.g. 'assembly line'
    op.reference                AS reference,         -- order / operation ref
    op.startdate::date          AS startdate,
    op.enddate::date            AS enddate,
    COALESCE(op.color, '#000')  AS barcolor           -- optional colour
FROM   public.operationplan           op
JOIN   public.resource                r  ON r.id = op.resource_id
WHERE  op.startdate >= '2019-06-01'
  AND  op.startdate <  '2019-07-01'
ORDER  BY r.name, op.startdate;
